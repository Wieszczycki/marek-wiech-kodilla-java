#lista zadań z tabeli ISSUES z dołączonymi nazwami list zadań z tabeli ISSUESLISTS:
SELECT I.SUMMARY AS TASK_SUMMARY, L.NAME AS LIST_NAME
FROM ISSUES I JOIN ISSUESLISTS L ON I.ISSUESLIST_ID = L.ID;

#lista zadań z tabeli ISSUES z przypisanymi do nich userami z tabeli USERS
SELECT I.SUMMARY AS TASK_SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

#imiona i nazwiska userów posiadających zadania z info o liczbie przypisanych zadań
SELECT U.ID, U.FIRSTNAME, U.LASTNAME, COUNT(*) AS LICZBA_ZADAN
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO;
